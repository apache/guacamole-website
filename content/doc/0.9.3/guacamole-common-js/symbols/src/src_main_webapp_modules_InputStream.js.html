 23 var Guacamole = Guacamole || {};
 24 
 25 /**
 26  * An input stream abstraction used by the Guacamole client to facilitate
 27  * transfer of files or other binary data.
 28  * 
 29  * @constructor
 30  * @param {Guacamole.Client} client The client owning this stream.
 31  * @param {Number} index The index of this stream.
 32  */
 33 Guacamole.InputStream = function(client, index) {
 34 
 35     /**
 36      * Reference to this stream.
 37      * @private
 38      */
 39     var guac_stream = this;
 40 
 41     /**
 42      * The index of this stream.
 43      * @type Number
 44      */
 45     this.index = index;
 46 
 47     /**
 48      * Called when a blob of data is received.
 49      * 
 50      * @event
 51      * @param {String} data The received base64 data.
 52      */
 53     this.onblob = null;
 54 
 55     /**
 56      * Called when this stream is closed.
 57      * 
 58      * @event
 59      */
 60     this.onend = null;
 61 
 62     /**
 63      * Acknowledges the receipt of a blob.
 64      * 
 65      * @param {String} message A human-readable message describing the error
 66      *                         or status.
 67      * @param {Number} code The error code, if any, or 0 for success.
 68      */
 69     this.sendAck = function(message, code) {
 70         client.sendAck(guac_stream.index, message, code);
 71     };
 72 
 73 };
