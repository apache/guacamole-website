 23 var Guacamole = Guacamole || {};
 24 
 25 /**
 26  * A reader which automatically handles the given input stream, returning
 27  * strictly text data. Note that this object will overwrite any installed event
 28  * handlers on the given Guacamole.InputStream.
 29  * 
 30  * @constructor
 31  * @param {Guacamole.InputStream} stream The stream that data will be read
 32  *                                       from.
 33  */
 34 Guacamole.StringReader = function(stream) {
 35 
 36     /**
 37      * Reference to this Guacamole.InputStream.
 38      * @private
 39      */
 40     var guac_reader = this;
 41 
 42     // Receive blobs as strings
 43     stream.onblob = function(data) {
 44 
 45         // Convert to string 
 46         var text = window.atob(data);
 47 
 48         // Call handler, if present
 49         if (guac_reader.ontext)
 50             guac_reader.ontext(text);
 51 
 52     };
 53 
 54     // Simply call onend when end received
 55     stream.onend = function() {
 56         if (guac_reader.onend)
 57             guac_reader.onend();
 58     };
 59 
 60     /**
 61      * Fired once for every blob of text data received.
 62      * 
 63      * @event
 64      * @param {String} text The data packet received.
 65      */
 66     this.ontext = null;
 67 
 68     /**
 69      * Fired once this stream is finished and no further data will be written.
 70      * @event
 71      */
 72     this.onend = null;
 73 
 74 };
